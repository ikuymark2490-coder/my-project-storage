รูปภาพใส่โค้ดไม่ได้เว้ย!

ระบบนี้ออกแบบให้รูปภาพเป็นเพียงไฟล์แสดงผลเท่านั้น ไม่รองรับการฝัง JavaScript หรือ Logic ภายในไฟล์รูปภาพ ถ้าต้องการให้รูปภาพทำงานได้ ให้ใช้ "ปุ่มกด" หรือ "กล่อง" แล้วเขียนโค้ดควบคุมแทน

เปลี่ยนข้อความ

เปลี่ยนคำบนปุ่มทันทีที่กด

this.innerText = "กดแล้ว";

ซ่อนปุ่ม / ลบทิ้ง

ทำให้ปุ่มหายไปหลังจากทำงานเสร็จ

this.style.display = "none";

สุ่มสีพื้นหลัง

กดหนึ่งทีเปลี่ยนสีหนึ่งครั้ง (Disco Mode)

const colors = ["#ef4444", "#22c55e", "#3b82f6", "#f59e0b", "#a855f7"];
const randomColor = colors[Math.floor(Math.random() * colors.length)];
this.style.backgroundColor = randomColor;

แจ้งเตือน Toast

ใช้ระบบแจ้งเตือนของ Engine แสดงข้อความ

showNotify("บันทึกข้อมูลเรียบร้อยแล้ว! 💾");

ปุ่มนับจำนวน (Counter)

กดเพื่อเพิ่มตัวเลขไปเรื่อยๆ

if(!window.count) window.count = 0;
window.count++;
this.innerText = "คะแนน: " + window.count;

เปิดเว็บไซต์อื่น

กดแล้วเด้งไปที่ URL ที่กำหนด

window.open("https://google.com", "_blank");

นาฬิกา Real-time

เปลี่ยนข้อความให้เป็นเวลาปัจจุบัน (อัปเดตทุกวินาที)

setInterval(() => {
  const now = new Date();
  this.innerText = now.toLocaleTimeString();
}, 1000);

สุ่มข้อความ (สุ่มคำคม)

กดแล้วสุ่มข้อความจากรายการที่กำหนด

const quotes = ["สวัสดี", "สู้ๆ นะ", "จินตนาการสำคัญกว่าความรู้", "อย่าหยุดฝัน"];
this.innerText = quotes[Math.floor(Math.random() * quotes.length)];

สลับความโปร่งใส (Fade Toggle)

คลิกที่กล่องเพื่อจางหายหรือแสดงขึ้นมา

this.style.transition = "opacity 0.5s";
this.style.opacity = (this.style.opacity == "0.3") ? "1" : "0.3";

ใส่ขอบเรืองแสง (Neon Glow)

เพิ่ม Shadow ให้กล่องดูเหมือนมีไฟนีออน

this.style.boxShadow = "0 0 20px #3b82f6";
this.style.border = "2px solid #3b82f6";
